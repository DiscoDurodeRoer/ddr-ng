import { Component, OnInit } from '@angular/core';
import { EvlSessionService, EvlToastService } from 'projects/evl-lib/src/public-api';

@Component({
  selector: 'app-tabs-showcase',
  templateUrl: './tabs-showcase.component.html',
  styleUrls: ['./tabs-showcase.component.scss']
})
export class TabsShowcaseComponent {

  constructor(
    private evlSession: EvlSessionService,
    private evlToast: EvlToastService  
  ) { 
    this.evlSession.setElement('title', 'Tabs'); 
  }

  tabIndex($event: number){
    this.evlToast.addSuccessMessage('Cambio de tab', 'Tab cambiado: ' + $event)
  }

}
